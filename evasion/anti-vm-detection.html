🖥️ Anti-VM Detection

MITRET1497
PlatformWindows
TypeVM Detection
DifficultyMedium

🧠 Description

Malware uses anti-VM techniques to detect and evade analysis in virtualized environments like VMware, VirtualBox, Hyper-V, and sandboxed systems.

📝 Registry-Based Detection

BOOL CheckVMRegistry() {
    const char* keys[] = {
        "SYSTEM\\CurrentControlSet\\Services\\VBoxGuest",
        "SYSTEM\\CurrentControlSet\\Services\\VBoxMouse",
        "SYSTEM\\CurrentControlSet\\Services\\VBoxService",
        "SYSTEM\\CurrentControlSet\\Services\\vmci",
        "SYSTEM\\CurrentControlSet\\Services\\vmhgfs",
        "SOFTWARE\\VMware, Inc.\\VMware Tools",
        "HARDWARE\\DEVICEMAP\\Scsi\\Scsi Port 0\\Scsi Bus 0\\Target Id 0\\Logical Unit Id 0",
        "HARDWARE\\DEVICEMAP\\Scsi\\Scsi Port 2\\Scsi Bus 0\\Target Id 0\\Logical Unit Id 0"
    };
    
    for (int i = 0; i < 8; i++) {
        HKEY hKey;
        if (RegOpenKeyEx(HKEY_LOCAL_MACHINE, keys[i], 0, KEY_READ, &hKey) == ERROR_SUCCESS) {
            RegCloseKey(hKey);
            return TRUE; // VM detected
        }
    }
    return FALSE;
}

        "C:\\Windows\\System32\\vboxtray.exe"
    };
    
    for (int i = 0; i < 11; i++) {
        if (GetFileAttributes(files[i]) != INVALID_FILE_ATTRIBUTES) {
            return TRUE; // VM detected
        }
    }
    return FALSE;
}

🔍 CPUID Detection

// CPUID with eax=1 checks Hypervisor bit
BOOL CheckCPUID() {
    int info[4];
    __cpuid(info, 1);
    // Bit 31 of ecx = Hypervisor bit
    if (info[2] & (1 << 31)) {
        // Check vendor string
        __cpuid(info, 0x40000000);
        char vendor[13];
        *(int*)vendor = info[1];
        *(int*)(vendor+4) = info[2];
        *(int*)(vendor+8) = info[3];
        vendor[12] = 0;
        
        if (strstr(vendor, "VMware") || 
            strstr(vendor, "VirtualBox") ||
            strstr(vendor, "Microsoft") ||
            strstr(vendor, "KVM") ||
            strstr(vendor, "QEMU"))
            return TRUE;
    }
    return FALSE;
}

// Check for Hyper-V signature
BOOL CheckHyperV() {
    int info[4];
    __cpuid(info, 0x40000000);
    if (info[1] == 0x20786974) return TRUE; // "Hy"
    return FALSE;
}

🌐 MAC Address Detection

BOOL CheckMAC() {
    const char* vm_macs[] = {
        "00:05:69", // VMware
        "00:0C:29", // VMware
        "00:1C:14", // VMware
        "00:50:56", // VMware
        "08:00:27", // VirtualBox
        "0A:00:27", // VirtualBox
        "52:54:00", // QEMU/KVM
        "00:16:3E", // Xen
        "00:1A:A0", // Hyper-V
        "00:03:FF", // Microsoft VPC
    };
    
    IP_ADAPTER_INFO adapter[16];
    DWORD size = sizeof(adapter);
    if (GetAdaptersInfo(adapter, &size) == 0) {
        PIP_ADAPTER_INFO pAdapter = adapter;
        while (pAdapter) {
            for (int i = 0; i < 10; i++) {
                if (strncmp(pAdapter->Address, vm_macs[i], 3) == 0)
                    return TRUE;
            }
            pAdapter = pAdapter->Next;
        }
    }
    return FALSE;
}

🏖️ Sandbox Detection

// Check for sandbox processes and files
BOOL CheckSandbox() {
    const char* processes[] = {
        "vmtoolsd.exe", "VBoxService.exe", "VBoxTray.exe",
        "sandboxiedll.dll", "snxhk.dll", "api-logger.dll",
        "CuckooHammer.exe", "inject.exe", "hookinject.exe"
    };
    
    // Check for specific modules
    HMODULE hMod = GetModuleHandle("sbiedll.dll");
    if (hMod) return TRUE;
    
    hMod = GetModuleHandle("dbghelp.dll");
    // Check if dbghelp is loaded (used by many sandboxes)
    
    // Check username
    char username[256];
    DWORD size = 256;
    GetUserName(username, &size);
    if (strstr(username, "sandbox") || strstr(username, "malware"))
        return TRUE;
    
    return FALSE;
}

// Check process count
BOOL CheckLowResources() {
    DWORD processes[1024], needed;
    EnumProcesses(processes, sizeof(processes), &needed);
    DWORD count = needed / sizeof(DWORD);
    
    if (count < 20) return TRUE; // Very few processes = sandbox
    return FALSE;
}
