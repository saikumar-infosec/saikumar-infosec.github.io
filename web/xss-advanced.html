Input Validation:

# HTML encoding (output encoding)
function escapeHtml(text) {
  const map = {
    '&': '&',
    '<': '<',
    '>': '>',
    '"': '"',
    "'": '''
  };
  return text.replace(/[&<>"']/g, c => map[c]);
}

# DOMPurify for HTML sanitization
import DOMPurify from 'dompurify';
